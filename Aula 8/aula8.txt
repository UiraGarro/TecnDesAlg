int soma(int x, int y){
    int res;
    res = x + y;
    return res;
}

float media(float num1, float num2){
    res = (num1 + num2) / 2;
}

int pot(int base, int expoente){
    int res = 1;
    for(int i = 0; i < expoente; i++){
        res = res * base;
    }
    return res;
}
